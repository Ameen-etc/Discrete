\begin{enumerate}[label=(\alph*)]
\item $2, 7, 12, \ldots$ to $10$ terms.
\item $-37, -33, -29, \ldots$ to $12$ terms.
\item $0.6, 1.7, 2.8, \ldots$ to $100$ terms.
\item $\frac{1}{15}, \frac{1}{12}, \frac{1}{10}, \ldots$ to $11$ terms.
\end{enumerate}
